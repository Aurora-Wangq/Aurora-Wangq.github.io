#include<iostream>  
#include<algorithm>  
using namespace std;

long int n,k,sum[10001],num=1,f[10001];  
long int i,j;

struct nk   
{  
    long int ks,js;  
};

nk z[10001];

int cmp(nk a,nk b) {

    return a.ks>b.ks;  
}

int main()  
{  

    scanf("%d%d",&n,&k);

    for(i=1;i<=k;i++) {

        cin>>z[i].ks>>z[i].js;    
        sum[z[i].ks]++;  
    }  

    sort(z+1,z+k+1,cmp);  

    for(i=n;i>=1;i--)  
    {  
        if(sum[i]==0)  
        f[i]=f[i+1]+1;  
        else for(j=1;j<=sum[i];j++)  
        {  
            if(f[i+z[num].js]>f[i])  
            f[i]=f[i+z[num].js];  
            num++;
        }  
    }  
    cout<<f[1]<<endl;  
}
