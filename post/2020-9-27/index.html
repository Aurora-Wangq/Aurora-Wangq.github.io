T15倍经验日

#include<iostream>
#include<cstdio>
#define ll long long
using namespace std;

ll n , v , lose[1002] , win[1002] , c[1002] , f[1002]; 

int main() {

    scanf("%lld%lld",&n,&v);

    for(int i=1;i<=n;i++) {

        scanf("%lld%lld%lld",&lose[i],&win[i],&c[i]);
        // win[i]*=5;
        // lose[i]*=5;
    }


       for (int i=1;i<=n;i++){
            for (int j=v;j>=c[i];j--){
                f[j]=max(f[j]+lose[i],f[j-c[i]]+win[i]);
            }
            for (int j=c[i]-1;j>=0;j--){
                f[j]+=lose[i];
            } 
        }

    printf("%lld",5*f[v]);

    return 0;
}

T2装箱

#include<iostream>
#include<cstdio>
using namespace std;

int cap , n , v[20004];  
int f[20004];

int main() {

    scanf("%d",&cap);
    scanf("%d",&n);

    for(int i=1;i<=n;i++) {

        scanf("%d",&v[i]);
    }

    for(int i=1;i<=n;i++) {

        for(int j=cap;j>=v[i];j--) {

            if(j>=v[i])
            f[j]=max(f[j],f[j-v[i]]+v[i]);
        }
    }

    printf("%d",cap-f[cap]);
    return 0;
}

T2开心的金明

#include<iostream>
#include<cstdio>

int n,m;
int v[30004],p[30004],f[30004];

using namespace std;
int main() {

    scanf("%d%d",&n,&m);

    for(int i = 1;i<=m;i ++) {

        scanf("%d%d",&v[i],&p[i]);
        p[i]*=v[i];
    }


    for(int i = 1;i<=m;i ++) {

        for(int j = n;j>=v[i];j --) {

            if(j>=v[i] ) {

                f[j]=max(f[j],f[j-v[i]]+p[i]);
            }
        }
    }

    printf("%d",f[n]);
    return 0;
}
