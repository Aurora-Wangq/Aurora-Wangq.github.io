#include<iostream>
#include<cstdio>
using namespace std;

int be[500004],tr[500004];
int n,m,opt,x,y;

int lowbit(int x) {

    return x&(-x);
}

void update(int pos,int w) {

    while(pos<=n) {

        tr[pos]+=w;
        pos+=lowbit(pos);
    }
} 

int fsum(int pos) {

    int ans=0;

    while(pos>0) {

        ans+=tr[pos];
        pos-=lowbit(pos);
    }

    return ans;
}

int main() {

    scanf("%d%d",&n,&m);

    for(int i=1;i<=n;i++) {

        scanf("%d",&be[i]);
//cout<<"21d12d"<<endl;
        update(i,be[i]);
    }

    for(int i=1;i<=m;i++) {

        scanf("%d",&opt);

        if(opt==1) {

            scanf("%d%d",&x,&y);

            update(x,y);
        }

        else {

            scanf("%d%d",&x,&y);

            printf("%d\n",fsum(y)-fsum(x-1));
        }
    }


    return 0;
}

#include<iostream>
#include<cstdio>
using namespace std;

int n,m,opt,x,y;
long long k,tr[500001],zc;

int lowbit(int a) {

    return a&(-a);
}

void updata(int pos,long long w) {

    while(pos<=n) {

        tr[pos]+=w;

        pos+=lowbit(pos);
    }
}

long long fsum(int x) {

    long long ans=0;

    while(x>0) {

        ans+=tr[x];

        x-=lowbit(x);
    }

    return ans;
}

int main() {

    long long now=0;

    scanf("%d%d",&n,&m);

    for(int i=1;i<=n;i++) {

        scanf("%lld",&zc);
        updata(i,zc-now);
        now=zc;
    }

    for(int i=1;i<=m;i++) {

        scanf("%d",&opt);

        if(opt==1) {

            scanf("%d%d%lld",&x,&y,&k);

            updata(x,k);
            updata(y+1,-k);
        }

        else {

            scanf("%d",&x);

            printf("%lld\n",fsum(x));
        } 
    }
    return 0;
}
