P4305 [JLOI2011]不重复数字

  1. 2020-10-18
  2. 389字
  3. 2 min read

哈希
思路 很简单

  1. 以二维vector为容器，第一维下标存储哈希值，第二维下标存储此哈希值下原数字的个数，数组值存储原数字。
  2. 输入一个数，找其对应哈希值下的数组内有没有和它相同的数字，相当于缩小了枚举的范围，因为原值相同的数哈希值一定相同，而不同的数哈希值也是可能相同的，所以要同时存储哈希值和原值。
  3. 如果没有重复的输出，有重复的不输出。
  4. 注意，由于是多组询问，每次都要把vector清零。
#include<iostream>
#include<cstdio>
#include<cmath>
#include<vector>
#define ll long long
using namespace std;

const int mod=100007;
const int maxn=5*1e5+7;
vector<int> h[mod];
int maxk,n,t;
ll tong[mod]; 

int hashe(int a) {return a=(a%mod+mod)%mod;}//哈希 

int find(int val) {
	
	int hx=hashe(val);
	maxk=max(maxk,hx);
	
	for(int i=0;i<h[hx].size();i++) {
		
		if(h[hx][i]==val) return 0;
	}
	
	h[hx].push_back(val);
		
	return 1;
}//判重

int main() {
	
	scanf("%d",&t);
	
	int num,n;
	
	for(int i=1;i<=t;i++) {
		
		scanf("%d",&n);
		
		for(int p=0;p<=maxk;p++) h[p].clear();//注意多次操作,要重新清零 
		maxk=0;
		
		for(int j=1;j<=n;j++) {
			
			scanf("%d",&num);
			
			if(find(num)==1) printf("%d ",num);  
		}
		
		printf("\n");
	}
	return 0;
}
