T1字符串哈希

#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
#define ull unsigned long long
using namespace std;

ull base=131;
ull a[10001];
char s[10001];
int n,ans=1;
int prime=23333; 
ull mod=1e9+7;

ull hax(char s[])
{

 int len=strlen(s);
 ull ans=0;
 for (int i=0;i<len;i++)
 ans=(ans*base+(ull)s[i])%mod+prime;
 return ans;
}

int main()
{

 scanf("%d",&n);
 for(int i=1;i<=n;i++) {

     scanf("%s",s);
     a[i]=hax(s);
}

 sort(a+1,a+n+1);

 for(int i=1;i<n;i++) {

     if(a[i]!=a[i+1])
     ans++;
 }

 printf("%d",ans);
 return 0;
}

#include<iostream>
#include<cstdio>
using namespace std;

int n;
int a[100100];

int main()
{

    scanf("%d",&n);

    int ans=-1;
    for (int i = 1;i <= n;i++) {

        scanf("%d",&a[i]);

        if (i != 1)
            ans=ans==-1?a[i-1]-a[i] : min(ans,a[i-1]-a[i]);    
    }
    printf("%d",ans);
}
