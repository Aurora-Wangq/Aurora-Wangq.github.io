T1[https://www.luogu.com.cn/problem/P1049

#include<iostream>
#include<cstdio>
using namespace std;
int f[20005],w[35];
int main()
{
    int v,n;
    cin>>v>>n;
    for(int i=1;i<=n;i++)cin>>w[i];
    for(int i=1;i<=n;i++)
        for(int j=v;j>=w[i];j--)
            f[j]=max(f[j],f[j-w[i]]+w[i]);
    cout<<v-f[v];
    return 0;
}

T2[https://www.luogu.com.cn/problem/P1359]

#include<iostream>
#include<cstdio>
using namespace std;

const int maxn=202;
int n,dis,f[maxn];

int main() {

    scanf("%d",&n);

    for(int i=1;i<=n;i++)
        for(int j=i+1;j<=n;j++)
        {
            scanf("%d",&dis);
            if(f[j]==0 || f[i]+dis<f[j]) f[j] = f[i]+dis;
        }

    printf("%d",f[n]);
    return 0;
}
