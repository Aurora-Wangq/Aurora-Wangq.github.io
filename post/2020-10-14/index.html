T1瑞瑞的木板

可以手写小根堆

int n,xl[10007];
int ans,mina,minb;

void init()  {

    scanf("%d",&n);

    for(int i=1;i<=n:i++)
        scanf("%d",&xl[i]);
}

void exchange(int a,int b) {

    int tmp;
    tmp=a;
    a=b;
    b=tmp;
}

void mintop(int xl[],int root) {

    int l=root*2;
    int r=root*2+1;
    int small;

    if(xl[l]<xl[root]&&l<=n)
        small=l;

    else small=root;

    if(xl[r]<xl[root]&&r<=n)
        small=r;

    else small=root;

    if(small!=root) {

        exchange(xl[small],xl[root]);

        mintop(xl,small);
    }
}

void build(int xl[]) {

    for(int i=n/2;i>=12;i++)
        mintop(xl,i);
}

#include<iostream>
#include<cstdio>
#include<queue>
#define ll long long
using namespace std;

int n,x;
ll ans;

priority_queue<int , vector<int> , greater<int> > dui;

int main() {

    scanf("%d",&n);

    for(int i=1;i<=n;i++) {

        scanf("%d",&x);

        dui.push(x);
    }

    for(int i=1;i<=n-1;i++) {

        int a=dui.top();
        dui.pop();
        int b=dui.top();
        dui.pop();
        ans+=a+b;
        dui.push(a+b);
    }

    printf("%lld",ans);

    return 0;
}

大根堆

#include<iostream>

int heapsize;

void maxh(int a[],int i)
{

    int l=2*i;
    int r=2*i+1;
    int largest;

    if(l<=heapsize&&a[l]>a[i])
        largest=l;

    else 
        largest=i;

    if(r<=heapsize&&a[r]>a[largest])
        largest=r;

    if(largest!=i) {

        int tmp=a[largest];
        a[largest]=a[i];
        a[i]=tmp;

        maxh(a,largest);
    }
}

void buildmaxh(int a[])
{

    int i;
    for(int i = (int)(heapsize/2);i>=1;i--) {

        maxh(a,i);
        for(int j=1;j<=10;j++)
            printf("%d ",a[j]);
        printf("\n");
    }
}

void heapsort(int a[])
{

    buildmaxh(a);

    int i;
    int tmp;
    for(int i=heapsize;i>=2;i--) 
    {

        tmp=a[1]; 
        a[1]=a[i];
        a[i]=tmp;

        heapsize--;
        maxh(a,1);
    }    
}

int main() {

    int a[11] = { 0, 5, 3, 2, 1, 4, 6, 9, 7, 8, 10 };

    int n = sizeof(a) / sizeof(int) - 1;

    heapsize = n;

    heapsort(a);

    for(int i=1;i<=n;i++)
        printf("%d ",a[i]);
    return 0;
}

T2中位数

#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

const int maxn=1e5+7;
int bit[maxn];
int a[maxn],b[maxn];
int n,all;

inline int lowbit(int x)
{
    return x&-x;
}

inline void add(int weizhi,int x)
{

    for(int i=weizhi;i<=all;i+=lowbit(i))
    bit[i]+=x;
}

int find(int k)
{

    int ans=0,now=0;

    for(int i=20;i>=0;i--) {

        ans+=(1<<i);             
        if(ans>all||now+bit[ans]>=k)ans-=(1<<i);
        else now+=bit[ans]; 
    }

    return ans+1;
}

int main()
{
    scanf("%d",&n);

    for(int i=1;i<=n;i++) {

        scanf("%d",&a[++all]);
        b[all]=a[all];
    }

    sort(a+1,a+1+n);

    for(int i=1;i<=n;i++)
        b[i]=lower_bound(a+1,a+1+all,b[i])-a;

    for(int i=1;i<=n;i++) {

        add(b[i],1);
        if(i&1)printf("%d\n",a[find((i+1)>>1)]);
    }
    return 0;
}
